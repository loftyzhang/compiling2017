procedure sorting (var num:array[5]of integer);
	var i,j,t:integer;
	begin
	for i=0 to 4
	do
		for j=i+1 to 4
		do
			if num[i]>num[j]
			then 
			begin
				t = num[i];
				num[i] = num[j];
				num[j] = t;
			end
	end; 
function getaverage (var num:array[5]of integer);
	var i:integer;
	var n:real;
